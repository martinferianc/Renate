6def average_accuracy(results: Dict[str, List[List[float]]], task_id: int) -> float: 

7 """Compute the average accuracy of a model. 

8 

9 This measure is defined by: 

10 

11 .. math:: 

12 \\frac{1}{T} sum_{i=1}^T a_{T,i} 

13 

14 where :math:`T` is the number of tasks, :math:`a_{T,i}` is the accuracy 

15 of the model on task :math:`i`, while having learned all tasks up to :math:`T`. 

16 

17 Args: 

18 results: The results dictionary holding all the results with respect to all recorded 

19 metrics. 

20 task_id: The task index. 

21 """ 

22 return sum(results["accuracy"][task_id][: task_id + 1]) / (task_id + 1)

25def forgetting(results: Dict[str, List[List[float]]], task_id: int) -> float: 

26 """Compute the forgetting measure of the model. 

27 

28 This measure is defined by: 

29 

30 .. math:: 

31 \\frac{1}{T-1} sum_{i=1}^{T-1} f_{T,i} 

32 

33 where :math:`f_{j,i}` is defined as: 

34 

35 .. math:: 

36 f_{j,i} = \\max_{k\\in\\{1, \\ldots, j=i\\}} a_{k,i} - a_{j,i} 

37 

38 where :math:`T` is the final task index, :math:`a_{n,i}` is the test classification accuracy 

39 on task :math:`i` after sequentially learning the nth task and :math:`f_{j,i}` is a measure of 

40 forgetting on task :math:`i` after training up to task :math:`j`. 

41 

42 Args: 

43 results: The results dictionary holding all the results with respect to all recorded 

44 metrics. 

45 task_id: The task index. 

46 """ 

47 if task_id == 0: 

48 return 0.0

50 def f(results: Dict[str, List[List[float]]], j: int, i: int) -> float: 

51 """A Helper function to compute the: math:`f_{j,i}`.""" 

52 accuracy_ji = results[j][i] 

53 max_accuracy_ki = max([results[k][i] for k in range(j)]) 

54 return max_accuracy_ki - accuracy_ji 

55 

56 sum_f = 0.0 

57 for i in range(task_id): 

58 sum_f += f(results["accuracy"], task_id, i) 

59 return sum_f / task_id

62def backward_transfer(results: Dict[str, List[List[float]]], task_id: int) -> float: 

63 """Compute the backward transfer measure of the model. 

64 

65 This measure is defined by: 

66 

67 .. math:: 

68 \\frac{1}{T-1} sum_{i=1}^{T-1} a_{T,i} - a_{i,i} 

69 

70 where :math:`T` is the final task index, :math:`a_{n,i}` is the test classification accuracy 

71 on task :math:`i` after sequentially learning the nth task. 

72 

73 Args: 

74 results: The results dictionary holding all the results with respect to all recorded 

75 metrics. 

76 task_id: The task index. 

77 """ 

78 if task_id == 0: 

79 return 0.0 

80 return ( 

81 sum([results["accuracy"][task_id][i] - results["accuracy"][i][i] for i in range(task_id)]) 

82 / task_id 

83 )

86def forward_transfer(results: Dict[str, List[List[float]]], task_id: int) -> float: 

87 """Compute the forward transfer measure of the model. 

88 

89 This measure is defined by: 

90 

91 .. math:: 

92 \\frac{1}{T-1} sum_{i=2}^{T} a_{i-1,i} - b_{i} 

93 

94 where :math:`T` is the final task index, :math:`a_{n,i}` is the test classification accuracy 

95 on task :math:`i` after sequentially learning the nth task. :math:`b_{i}` is the accuracy for 

96 all :math:`T` tasks recorded at initialisation prior to observing any task. 

97 

98 Args: 

99 results: The results dictionary holding all the results with respect to all recorded 

100 metrics. 

101 task_id: The task index. 

102 """ 

103 if task_id == 0: 

104 return 0.0 

105 return sum( 

106 [ 

107 results["accuracy"][i - 1][i] - results["accuracy_init"][0][i] 

108 for i in range(1, task_id + 1) 

109 ] 

110 ) / (task_id)
