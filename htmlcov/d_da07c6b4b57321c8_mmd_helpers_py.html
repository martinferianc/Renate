10def mmd_gram( 

11 K: torch.Tensor, z: torch.Tensor, num_permutations: int = 0 

12) -> Tuple[torch.Tensor, Optional[torch.Tensor]]: 

13 """Maximum mean discrepancy based on a precomputed kernel-Gram matrix. 

14 

15 This computes the test statistic and (optionally) p-value to conduct an MMD two-sample test to 

16 decide whether two sets are generated by the same distribution. The inputs are passed implicitly 

17 in the form of a kernel Gram matrix, evaluated across the union of both sets, and a binary 

18 vector indicating the assignments of data points to the two sets. I.e., a value of `z[i] = 0` 

19 indicates that the `i`-th data point belongs to set zero. Optionally, a permutation test is 

20 carried out and a p-value is returned alongside the raw test statistic. 

21 

22 MMD tests have been proposed by 

23 

24 [1] Gretton, A., et al. A kernel two-sample test. JMLR (2012). 

25 

26 Args: 

27 K: A tensor containing a kernel-Gram matrix (size `(n, n)`). 

28 z: A binary vector of length `n` indicating the partition. 

29 num_permutations: If this is a positive number, a permutation test will be carried out and 

30 an approximate p-value will be returned. 

31 

32 Returns: 

33 A tuple `(t, p)` of two scalar floats, where `t` is the value of the MMD test statistic and 

34 `p` is the p-value (or `None` if `num_permutations=0`). 

35 """ 

36 n = K.size(0) 

37 assert K.size(1) == n 

38 assert z.size() == (n,) 

39 inds_0 = torch.where(z == 0)[0] 

40 inds_1 = torch.where(z == 1)[0] 

41 n0 = len(inds_0) 

42 n1 = len(inds_1) 

43 

44 mmd = ( 

45 K[inds_0][:, inds_0].sum() / (n0 * (n0 - 1)) 

46 + K[inds_1][:, inds_1].sum() / (n1 * (n1 - 1)) 

47 - 2 * K[inds_0][:, inds_1].mean() 

48 ) # MMD statistic, see Eq. (5) in [1]. 

49 

50 if num_permutations == 0: 

51 return mmd, None 

52 

53 # Permutation test: Randomize the assignments z, compute MMD, and count how often we exceed the 

54 # value obtained with original z. 

55 cnt = 0 

56 for _ in range(num_permutations): 

57 z_ = torch.zeros(n, device=K.device) 

58 pi = torch.randperm(n, device=K.device) 

59 z_[pi[:n1]] = 1 

60 mmd0, _ = mmd_gram(K, z_, num_permutations=0) 

61 if mmd0 > mmd: 

62 cnt += 1 

63 p_val = torch.tensor(cnt / num_permutations) 

64 

65 return mmd, p_val

68def mmd( 

69 X0: torch.Tensor, X1: torch.Tensor, kernel: Kernel, num_permutations: int = 0 

70) -> Tuple[torch.Tensor, Optional[torch.Tensor]]: 

71 """Compute MMD between two samples. 

72 

73 Optionally, return an estimated p-value based on a permutation test. 

74 

75 Args: 

76 X0: First sample, shape (n0, dx). 

77 X1: Second sample, shape (n1, dx). 

78 kernel: Kernel function to use. 

79 num_permutations: If this is a positive number, a permutation test will 

80 be carried out and a p-value will be returned. 

81 

82 Returns: 

83 A tuple `(t, p)` where `t` is the value of the MMD test statistic and `p` is the p-value 

84 (or `None` if `num_permutations=0`). 

85 """ 

86 X = torch.cat([X0, X1], dim=0) 

87 z = torch.cat([torch.zeros(X0.size(0)), torch.ones(X1.size(0))], dim=0) 

88 K = kernel(X, X) 

89 assert not torch.any(torch.isnan(K)) 

90 return mmd_gram(K, z, num_permutations)
